# 904. Fruit Into Baskets

```cpp
class Solution {
public:
    int totalFruit(vector<int>& fruits) {
        int maxTreeTotal = 0;

        int mainTreeType = -1;
        int otherTreeType = -1; 
        
        int mainTreeCount = 0;     
        int currentTreeTotal = 0;
        
        for(auto tree: fruits) {             
            if(tree == mainTreeType or tree == otherTreeType) {
                currentTreeTotal++;
            } else {
                currentTreeTotal = mainTreeCount + 1;
            }
            
            maxTreeTotal = max(maxTreeTotal, currentTreeTotal);
      
            if(tree == mainTreeType) {
                mainTreeCount++;
            } else {
                otherTreeType = mainTreeType;
                mainTreeType = tree;
                mainTreeCount = 1;
            }
        }
        
        return maxTreeTotal;
    }
};
```
